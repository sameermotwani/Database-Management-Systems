Sample Exercise

Simple example of a query using an enumeration. 

create table Evaluation(
  id int primary key,
  evaluator varchar(200) not null,
  evaluation enum('poor', 'average', 'good', 'excellent') not null
);
insert into Evaluation values
  (1, 'Fred', 'good'),
  (2, 'George', 'average'),
  (3, 'Mary', 'excellent');
select e.evaluation+0
  from Evaluation e;
